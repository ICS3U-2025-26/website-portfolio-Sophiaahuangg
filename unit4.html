Final Project: Cake Smashing Game

My Coding Process and Design Reflections: This is my final culminating game for the unit, called Cake Smash! My design process focused on building an interactive arcade game styled like Whac-A-Mole, but using custom cake assets instead. I wanted to make the game fun for everyone, so I created a difficulty menu at the start using a gameState controller variable that switches between the title screen, level selection, the active match, and the game over panel.

The trickiest part of the coding process was managing the cake spawning timers across the different difficulty tiers. I used an object array to hold values for Easy, Medium, and Hard speeds. To keep track of individual cakes popping up from the ground holes, I used the millis() clock engine. If a cake stays active on screen for too long without the player clicking it, the code automatically removes the cake asset, adds to the miss counter, and shifts our HUD display heart vectors. It took a lot of debugging to prevent cakes from spawning over menu boxes, but the state machine works perfectly now!

// GAME STATE
let gameState = "title";
let currentLevel = "easy";
let levels = {
  easy:   { cakeTime: 1500, spawnInterval: 1000, maxActive: 3 },
  medium: { cakeTime: 1000, spawnInterval: 700,  maxActive: 4 },
  hard:   { cakeTime: 700,  spawnInterval: 500,  maxActive: 5 }
};

let CAKE_TIME = 1500;
let SPAWN_INTERVAL = 1000;
let MAX_ACTIVE_CAKES = 3;

// IMAGES
let startImg;
let titleImg;
let bgImg;
let cakeImgs = [];

// SOUNDS
let crunchSound;
let missSound;
let gameOverSound;
let levelUpSound;

// BUTTONS
let startButton = { x: 0, y: 0, w: 200, h: 70 };
let playAgainButton = { x: 0, y: 260, w: 250, h: 60 };
let levelButtons = [];

// GAME DATA
let score = 0;
let misses = 0;
const MAX_MISSES = 3;
let holes = [];
const HOLE_RADIUS = 35;
let nextSpawnTime = 0;

// load images and assets
function preload() {
  startImg = loadImage("images/StartGame.png");
  titleImg = loadImage("images/cakesmashing.png");
  bgImg = loadImage("images/background.png");
  cakeImgs[0] = loadImage("images/cake1.png");
  cakeImgs[1] = loadImage("images/cake2.png");
  cakeImgs[2] = loadImage("images/cake3.png");
  cakeImgs[3] = loadImage("images/cake4.png");
  cakeImgs[4] = loadImage("images/cake5.png");
  cakeImgs[5] = loadImage("images/cake6.png");
  
  crunchSound = loadSound("sounds/crunch.mp3");
  missSound = loadSound("sounds/miss.mp3");
  gameOverSound = loadSound("sounds/gameover.mp3");
  levelUpSound = loadSound("sounds/levelup.mp3");
}

function setup() {
  createCanvas(600, 400);
  textAlign(CENTER, CENTER);
  buildHoles();
  startButton.x = width / 2 - 100;
  startButton.y = height * 0.55;
  playAgainButton.x = width / 2 - 125;
  levelButtons = [
    { label: "Easy",   level: "easy",   x: width/2 - 125, y: 180, w: 250, h: 50 },
    { label: "Medium", level: "medium", x: width/2 - 125, y: 240, w: 250, h: 50 },
    { label: "Hard",   level: "hard",   x: width/2 - 125, y: 300, w: 250, h: 50 }
  ];
}

function draw() {
  background(255);
  if (gameState === "title") { drawTitleScreen(); }
  else if (gameState === "levelselect") { drawLevelSelectScreen(); }
  else if (gameState === "play") { drawPlayScreen(); }
  else if (gameState === "gameover") { drawGameOverScreen(); }
}

function drawTitleScreen() {
  image(titleImg, 0, 0, width, height);
  fill(255, 235, 245, 220);
  rect(50, height - 120, width - 100, 80, 10);
  fill(120, 30, 70);
  textSize(14);
  text("HOW TO PLAY:", width/2, height - 100);
  textSize(12);
  text("Click on cakes before they disappear!", width/2, height - 80);
  text("Miss 3 cakes and it's GAME OVER!", width/2, height - 60);
  image(startImg, startButton.x, startButton.y, startButton.w, startButton.h);
}

function drawLevelSelectScreen() {
  image(bgImg, 0, 0, width, height);
  fill(0); textSize(36); text("Choose Difficulty", width / 2, 100);
  for (let b of levelButtons) {
    fill(255, 140, 180); rect(b.x, b.y, b.w, b.h, 12);
    fill(255); textSize(18); text(b.label, b.x + b.w/2, b.y + b.h/2);
  }
}

function mousePressed() {
  if (gameState === "title" && isInsideButton(startButton)) {
    gameState = "levelselect"; return;
  }
  if (gameState === "levelselect") {
    for (let b of levelButtons) {
      if (isInsideButton(b)) {
        currentLevel = b.level; applyLevelSettings(); resetGame(); return;
      }
    }
  }
  if (gameState === "gameover" && isInsideButton(playAgainButton)) {
    gameState = "levelselect"; return;
  }
  if (gameState === "play") {
    for (let h of holes) {
      if (!h.active) continue;
      let d = dist(mouseX, mouseY, h.x, h.y);
      if (d < HOLE_RADIUS) {
        score++; h.active = false; crunchSound.play();
      }
    }
  }
}

function isInsideButton(button) {
  return (mouseX > button.x && mouseX < button.x + button.w && mouseY > button.y && mouseY < button.y + button.h);
}

function applyLevelSettings() {
  let s = levels[currentLevel];
  CAKE_TIME = s.cakeTime; SPAWN_INTERVAL = s.spawnInterval; MAX_ACTIVE_CAKES = s.maxActive;
}

function drawPlayScreen() {
  image(bgImg, 0, 0, width, height);
  fill(230, 160, 190, 120); rect(0, height - 40, width, 40);
  spawnCakes(); checkExpiredCakes(); drawHoles(); drawHUD();
}

function buildHoles() {
  holes = []; let spacingX = 150; let spacingY = 110;
  let startX = width / 2 - spacingX; let startY = 140; let index = 0;
  for (let row = 0; row < 2; row++) {
    for (let col = 0; col < 3; col++) {
      holes.push({ x: startX + col * spacingX, y: startY + row * spacingY, active: false, timer: 0, img: cakeImgs[index] });
      index++;
    }
  }
}

function spawnCakes() {
  if (millis() < nextSpawnTime) return;
  nextSpawnTime = millis() + SPAWN_INTERVAL;
  let active = holes.filter(h => h.active).length;
  if (active >= MAX_ACTIVE_CAKES) return;
  let inactive = holes.filter(h => !h.active);
  if (inactive.length === 0) return;
  let chosen = random(inactive); chosen.active = true; chosen.timer = millis();
}

// EXPIRATION MISS CHECKER
function checkExpiredCakes() {
  for (let h of holes) {
    if (h.active && millis() - h.timer > CAKE_TIME) {
      h.active = false; misses++;
      if (misses >= MAX_MISSES) { gameState = "gameover"; }
    }
  }
}

function drawHoles() {
  for (let h of holes) {
    if (h.active) {
      let size = HOLE_RADIUS * 3; imageMode(CENTER); image(h.img, h.x, h.y, size, size);
    } else {
      noStroke(); fill(60, 20, 40); ellipse(h.x, h.y, HOLE_RADIUS * 2);
    }
  }
  imageMode(CORNER);
}

function drawHUD() {
  fill(255, 170, 200, 210); rect(0, 0, width, 40);
  fill(120, 30, 70); textSize(16); textAlign(LEFT, CENTER); text("Score: " + score, 15, 20);
  textAlign(RIGHT, CENTER);
  for (let i = 0; i < MAX_MISSES; i++) {
    if (i < MAX_MISSES - misses) fill(220, 60, 100); else fill(200, 160, 175);
    text("♥", width - 15 - i * 30, 20);
  }
  textAlign(CENTER, CENTER); text("Level: " + currentLevel.toUpperCase(), width / 2, 20);
}

function drawGameOverScreen() {
  image(bgImg, 0, 0, width, height);
  fill(180, 40, 80); textSize(40); text("Game Over!", width / 2, 70);
  let boxW = 300; let boxH = 120; let boxX = width / 2 - boxW / 2; let boxY = 110;
  fill(255, 235, 245); rect(boxX, boxY, boxW, boxH, 18);
  fill(120, 30, 70); textSize(18); text("Final Score", width / 2, boxY + 25);
  fill(200, 50, 90); textSize(42); text(score, width / 2, boxY + 65);
  textSize(14); text("Misses: " + misses + "/" + MAX_MISSES, width / 2, boxY + 95);
  playAgainButton.x = width / 2 - playAgainButton.w / 2; playAgainButton.y = 260;
  fill(255, 140, 180); rect(playAgainButton.x, playAgainButton.y, playAgainButton.w, playAgainButton.h, 14);
  fill(255); textSize(20); text("Play Again", width / 2, playAgainButton.y + playAgainButton.h / 2);
}

function resetGame() {
  score = 0; misses = 0; nextSpawnTime = millis() + 500; buildHoles(); gameState = "play";
}
